// Comments will NOT be generated

open logicFuncs

sig state{
	arg1:Int,		// int x
	arg2:Int		// int y
}

// parag1 is int x before if statement
// parag2 is int y before if statement
fun funStatement0 (parg1:Int,parg2:Int) : state {
	{ n : state | 
		n.arg1 = ((parg1 > 0 => True else False) in True => 3 else parg2.add[2] )
		 and n.arg2 = ((parg1 > 0 => True else False) in True => parg2 else parg2 )
		 
	}
}

assert assertStatement0 {
	 all n: state | (n.arg1 = 1 => True else False) in True => (all field: funStatement0 [n.arg1,n.arg2] | { field.arg1 > 2 => True else False}  in True)
}

// check that the assertion holds (it should hold in this case)
check assertStatement0
